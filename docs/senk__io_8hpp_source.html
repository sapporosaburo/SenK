31 std::getline(ifs, line);
32 if(line.find("complex") != std::string::npos) {
33 std::cerr << "# Header line is not valid." << std::endl;
34 return false;
35 }
36 if(line.find("symmetric") != std::string::npos) { shape[0] = Sym; }

38 else { std::cerr << "# Header line is not valid." << std::endl; return false; }
39
40 int PE;
41 while(std::getline(ifs, line)) {
42 if(line[0] == '%') { continue; }
43 std::istringstream iss(line);
44 iss >> N[0] >> M[0] >> PE;
45 break;

47 std::cout << "# " << N[0] << " " << M[0] << " " << PE << std::endl;
48 *val = utils::SafeMalloc<double>(PE); //new double[PE];

50 *rptr = utils::SafeMalloc<int>(N[0]+1); //new int[N[0]+1];
51 int *row = utils::SafeMalloc<int>(PE); //new int[PE];
52
53 int nnz = 0;
54 double t_val;
55 int t_row, t_col;
56 for(int i=0; i<PE; i++) {
57 ifs >> t_row >> t_col >> t_val;
58 if(removeZeros && t_val == 0) continue;
59 (*val)[nnz] = t_val;
60 row[nnz] = t_row-1;
61 (*cind)[nnz] = t_col-1;
62 nnz++;
63 }
64 if(nnz != PE) {
65 *val = utils::SafeRealloc<double>(*val, nnz);
66 *cind = utils::SafeRealloc<int>(*cind, nnz);
67 }
68 helper::QuickSort<int, int, double>(row, *cind, *val, 0, nnz-1);
69 int cnt = 0;
70 (*rptr)[0] = 0;
71 for(int i=0; i<N[0]; i++) {
72 int num = 0;
73 while(row[cnt] == i) {
74 num++; cnt++;
75 if(cnt == nnz) { break; }
76 }
77 (*rptr)[i+1] = (*rptr)[i] + num;
78 helper::QuickSort<int, double>(*cind, *val, (*rptr)[i], (*rptr)[i+1]-1);
79 }
80 std::free(row);
81 return true;
82}
